        input, select, button { width: 100%; padding: 10px; margin-top: 5px; box-sizing: border-box; }
        table { width: 100%; border-collapse: collapse; margin-top: 20px; }
        th, td { border: 1px solid #ddd; padding: 8px; text-align: center; }

    </style>

<body>
    <h2>記帳本</h2>
    <div class="form-group">
        <select id="person">
            <option value="家佑">家佑</option>
            <option value="梅心">梅心</option>
        </select>
        <input type="text" id="item" placeholder="項目名稱">
        <input type="number" id="amount" placeholder="金額">
        <button onclick="addEntry()">新增紀錄</button>
    </div>
    <div class="summary" id="summary"></div>
    <table>
        <thead><tr><th>人</th><th>項目</th><th>金額</th><th>操作</th></tr></thead>
        <tbody id="list"></tbody>
    </table>
    <script>
        let data = JSON.parse(localStorage.getItem('expenses')) || [];
        function updateUI() {
            const list = document.getElementById('list');
            const summary = document.getElementById('summary');
            list.innerHTML = '';
            let totalJiayou = 0, totalMeixin = 0;
            data.forEach((entry, index) => {
                if(entry.person === '家佑') totalJiayou += entry.amount;
                else totalMeixin += entry.amount;
                list.innerHTML += `<tr><td>${entry.person}</td><td>${entry.item}</td><td>${entry.amount}</td>
                <td><button onclick="deleteEntry(${index})">刪</button></td></tr>`;
            });
            const total = totalJiayou + totalMeixin;
            const avg = total / 2;
            const diff = totalJiayou - avg;
            
            let result = `總支出: ${total}元 | 平均每人負擔: ${avg}元<br>`;
            if (diff > 0) result += `<strong>梅心應付給家佑 ${diff} 元</strong>`;
            else if (diff < 0) result += `<strong>家佑應付給梅心 ${Math.abs(diff)} 元</strong>`;
            else result += `<strong>目前已平衡</strong>`;
            
            summary.innerHTML = result;
            localStorage.setItem('expenses', JSON.stringify(data));
        }
        function addEntry() {
            const person = document.getElementById('person').value;
            const item = document.getElementById('item').value;
            const amount = parseInt(document.getElementById('amount').value);
            if(!item || !amount) return alert("請輸入完整資訊");
            data.push({person, item, amount});
            updateUI();
        }
        function deleteEntry(index) {
            data.splice(index, 1);
            updateUI();
        }
        updateUI();
    </script>
</body>
